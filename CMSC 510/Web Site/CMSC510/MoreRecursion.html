  if k == 0 return false

Print Permutations

#include <iostream>
#include <vector>
#include <fstream>

void printSet(const std::vector<int>& A, std::ofstream& out) {
  out << '[';
  auto itr = A.rbegin();
  out << *itr;
  for (itr++; itr != A.rend(); itr++)
    out << ',' << *itr;
  out << ']' << std::endl;
}

void swap(std::vector<int>& A, int i, int j) {
  if (i != j) {
    int temp = A[i];
    A[i] = A[j];
    A[j] = temp;
  }
}

void permute(std::vector<int>& A, int k, std::ofstream& out) {
  if (k == A.size() - 1)
    printSet(A,out);
  else {
    for (int j = k; j < A.size(); j++) {
      swap(A, j, k);
      permute(A, k + 1,out);
      swap(A, j, k);
    }
  }
}

void printPermutations(std::vector<int>& A, std::ofstream& out) {
  permute(A, 0, out);
}

int main()
{
  std::vector<int> V;
  for (int n = 1; n <= 6; n++)
    V.push_back(n);

  std::ofstream out;
  out.open("permutations.txt");

  printPermutations(V, out);

  out.close();
  return 0;
}

Print All Subsets of a Certain Size

#include <iostream>
#include <vector>
#include <fstream>

void printSet(const std::vector<int>& A, std::ofstream& out) {
  out << '[';
  auto itr = A.rbegin();
  out << *itr;
  for (itr++; itr != A.rend(); itr++)
    out << ',' << *itr;
  out << ']' << std::endl;
}

void printHelper(std::vector<int> &V, int k, std::vector<int> &S, std::ofstream& out) {
  if (k == 0)
    printSet(S,out);
  else if (k <= V.size()) {
    int last = V.back();
    V.pop_back();
    // Two alternatives: use last or don't use last
    // Alternative one: don't use last
    printHelper(V, k, S, out);
    // Alternative two: do use last
    S.push_back(last);
    printHelper(V, k - 1, S, out);
    S.pop_back();
    V.push_back(last);
  }
}

void printAllSubsets(std::vector<int> &V, int k,std::ofstream& out) {
  std::vector<int> S;
  printHelper(V, k, S,out);
}

int main()
{
  std::vector<int> V;
  for (int n = 1; n <= 8; n++)
    V.push_back(n);

  std::ofstream out;
  out.open("subsets.txt");

  printAllSubsets(V, 5, out);

  out.close();
  return 0;
}
